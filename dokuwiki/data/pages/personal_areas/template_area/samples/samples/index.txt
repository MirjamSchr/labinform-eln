===== Create new sample entry =====

<form>
Action template :personal_areas:@@short@@:samples:samples:template_sample :personal_areas:@@short@@:samples:samples:@@No@@
labels labels

Fieldset          "New entry for samples"
hiddenautoinc     "No"
static            "Necessary fields"
textbox           "Description"
date              "Date"
data_aliastextbox "Batch" _pagebatch_@@short@@
static            "Optional fields"
data_aliastextbox "Container" _pagecontainer !
data_aliastextbox "Cap" _sacap !
data_aliastextbox "Substrate" _pagesubstrate !
textbox           "Concentration" !
data_aliastextbox "Measured" _bool ! =false
data_aliastextbox "State" _state ! =exists
data_aliastextbox "Storage" _storage ! =RT
thanks            "Thanks for your entry."
submit            "Create entry"
</form>
